# UndefinedBehaviorSanitizer suppressions for clink.
#
# Format: <check>:<source-path-substring> (one rule per line, # for comments).
# Loaded via UBSAN_OPTIONS=suppressions=...:print_suppressions=1.
#
# Like tsan-suppressions.txt: we only suppress reports that originate
# inside third-party libraries we vendor - not anything in clink's
# own code. UBSan reports against vendored code are upstream's
# concern, not ours, and surface as test failures that mask real
# issues we *should* catch.

# RocksDB v11.1.1 ships a hand-tuned ARM64 CRC32C implementation in
# util/crc32c_arm64.cc that loads 8 bytes from a not-necessarily-
# 8-byte-aligned address. Compiles down to a NEON sequence that
# tolerates the misalignment, but UBSan's `alignment` check fires
# on the raw load. Well-known upstream issue. Suppress for now -
# the fix lives in RocksDB.
alignment:rocksdb-src/

# RocksDB's port/port_posix and util layers occasionally hit
# `function` (cast to mismatched function pointer type) and
# `vptr` (member call on dynamic-type mismatch) checks for the same
# reason - vendored code we don't instrument cleanly.
function:rocksdb-src/
vptr:rocksdb-src/

# Apache Arrow's columnar arrays load multi-byte values (offsets in
# array_binary.h, primitives in array_primitive.h) straight out of
# packed buffers whose slices are not guaranteed 4/8-byte aligned.
# This is by design - Arrow's format permits unaligned access and its
# accessors compile to instructions that tolerate it - but UBSan's
# `alignment` check fires on the raw load. Every report is inside the
# system Arrow headers (/usr/include/arrow/...), never clink code;
# clink's own Arrow code lives under include/clink/, so this scope
# cannot hide a clink alignment bug. Surfaces via the Arrow-IPC wire
# path (NetworkChannel) and the S3 object cache. Same third-party
# class as the libarrow TSan suppressions.
alignment:include/arrow/
